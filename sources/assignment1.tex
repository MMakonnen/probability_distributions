\begin{exercise}

\begin{minted}[]{python}
import numpy as np
from scipy.stats import expon
from scipy.integrate import quad

labda = 3

X = expon(scale=1 / labda).rvs(1000)
# print(X)
print(X.mean())

success = (X > 1) * (X < 4)
# print(success)
print(success.mean(), success.std())


def F(x):  # CDF
    return 1 - np.exp(-labda * x)


def f(x):  # density
    return labda * np.exp(-labda * x)


print(F(4) - F(1))

I = quad(f, 1, 4)
print(I)
\end{minted}
